<script>
function togglePassword(id) {
  const input = document.getElementById(id);
  input.type = input.type === "password" ? "text" : "password";
}

/* Clear form data (security) */
function clearForm(formId) {
  const form = document.getElementById(formId);
  const inputs = form.querySelectorAll("input, select");
  inputs.forEach(el => {
    if (el.type === "checkbox") el.checked = false;
    else el.value = "";
  });
}

function switchToRegister() {
  clearForm("loginForm");
  loginForm.style.display = "none";
  registerForm.style.display = "block";
  title.innerText = "Register";
}

function switchToLogin() {
  clearForm("registerForm");
  registerForm.style.display = "none";
  loginForm.style.display = "block";
  title.innerText = "Login";
}

/* LOGIN VALIDATION */
function login() {
  let valid = true;
  loginEmailError.innerText = "";
  loginPasswordError.innerText = "";
  loginCaptchaError.innerText = "";

  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

  if (!emailRegex.test(loginEmail.value.trim())) {
    loginEmailError.innerText = "* Enter a valid email";
    valid = false;
  }

  if (loginPassword.value.trim() === "") {
    loginPasswordError.innerText = "* Password is required";
    valid = false;
  }

  if (!loginCaptcha.checked) {
    loginCaptchaError.innerText = "* Please verify captcha";
    valid = false;
  }

  if (valid) alert("Login Successful ‚úÖ");
}

/* REGISTER VALIDATION */
function register() {
  let valid = true;

  regNameError.innerText = "";
  regEmailError.innerText = "";
  regPasswordError.innerText = "";
  regConfirmError.innerText = "";
  regRoleError.innerText = "";
  regResumeError.innerText = "";
  regCaptchaError.innerText = "";

  if (!/^([A-Za-z]+)\s([A-Za-z]+)\s([A-Za-z]+)$/.test(regName.value.trim())) {
    regNameError.innerText = "* Enter First Middle Last name";
    valid = false;
  }

  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(regEmail.value.trim())) {
    regEmailError.innerText = "* Enter a valid email";
    valid = false;
  }

  const passRegex =
    /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&]).{8,}$/;
  if (!passRegex.test(regPassword.value)) {
    regPasswordError.innerText =
      "* Min 8 chars, uppercase, lowercase, number & symbol";
    valid = false;
  }

  if (regPassword.value !== regConfirm.value) {
    regConfirmError.innerText = "* Passwords do not match";
    valid = false;
  }

  if (regRole.value === "") {
    regRoleError.innerText = "* Select an interview role";
    valid = false;
  }

  if (!regResume.value) {
    regResumeError.innerText = "* Upload your resume";
    valid = false;
  }

  if (!regCaptcha.checked) {
    regCaptchaError.innerText = "* Please verify captcha";
    valid = false;
  }

  if (valid) alert("Registration Successful üéâ");
}

<script>
/* üîê STRICT LOGIN VALIDATION (ADDED ONLY) */
function login() {
  let valid = true;

  loginEmailError.innerText = "";
  loginPasswordError.innerText = "";
  loginCaptchaError.innerText = "";

  const email = loginEmail.value.trim();
  const password = loginPassword.value;

  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  const strongPassRegex =
    /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&]).{8,}$/;

  /* Email check */
  if (!emailRegex.test(email)) {
    loginEmailError.innerText = "* Enter a valid email address";
    valid = false;
  }

  /* Password strength check */
  if (!strongPassRegex.test(password)) {
    loginPasswordError.innerText =
      "* Min 8 chars, uppercase, lowercase, number & symbol";
    valid = false;
  }

  /* Captcha check */
  if (!loginCaptcha.checked) {
    loginCaptchaError.innerText = "* Please verify captcha";
    valid = false;
  }

  if (valid) {
    alert("Login Successful ‚úÖ");
  }
}
